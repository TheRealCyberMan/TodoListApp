document.addEventListener('DOMContentLoaded', function () { const sendButton = document.getElementById('sendBtn'); const messageInput = document.getElementById('messageInput'); const chatlogs = document.getElementById('chatlogs'); sendButton.addEventListener('click', function() { let messageContent = messageInput.value.trim(); if (messageContent !== '') { // Create a message element for the user message let userMessageElement = document.createElement('div'); userMessageElement.className = 'user-message'; userMessageElement.textContent = messageContent; chatlogs.appendChild(userMessageElement); // Send the user message to the server for processing sendMessageToServer(messageContent); } }); // Function to send user messages to the server function sendMessageToServer(message) { fetch('/chat', { method: 'POST', body: JSON.stringify({ message: message }), headers: { 'Content-Type': 'application/json' } }) .then(response => response.json()) .then(data => { // Create a message element for the chatbot's response let chatbotMessageElement = document.createElement('div'); chatbotMessageElement.className = 'chatbot-message'; chatbotMessageElement.textContent = data.response; chatlogs.appendChild(chatbotMessageElement); // Clear the input field for new messages messageInput.value = ''; }); } });
